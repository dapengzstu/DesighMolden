
public class IntegerDemo
{

    public static void main(String[] args)
    {
        Integer a1 = 127;
        Integer a2 = 127;

        System.out.println(a1 == a2);

        Integer a3 = 129;
        Integer a4 = 129;

        System.out.println(a3 == a4);
    }
}
结果输出 true，false
